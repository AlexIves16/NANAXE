# Firebase configuration file
# Download this from Firebase Console:
# Project Settings > Your apps > Add app

# For Android:
# 1. Add your app with package name: com.nanahui.nanahui_crm
# 2. Download google-services.json
# 3. Place it in: android/app/google-services.json

# For iOS:
# 1. Add your app with bundle ID: com.nanahui.nanahuiCrm
# 2. Download GoogleService-Info.plist
# 3. Place it in: ios/Runner/GoogleService-Info.plist

# For Web:
# 1. Add web app
# 2. Copy firebaseConfig object
# 3. Update lib/core/firebase_options.dart with your config

# Required Firebase APIs:
# - Authentication (Google Sign-In)
# - Firestore Database
# - Cloud Messaging (FCM)
# - Cloud Functions (optional, for AI processing)
